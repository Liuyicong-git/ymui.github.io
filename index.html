# 快速上手

# 完全引入

你可以引入整个 Element，或是根据需要仅引入部分组件。我们先介绍如何引入完整的 Element。

 import Vue from 'vue'
 import YMSUI from 'yms-ui';
 import "yms-ui/lib/style/index.css";
 import App from './App.vue'

 Vue.use(YMSUI)

 new Vue({
   el: '#app',
   render: h => h(App)
 })

# 按需引入

  import Vue from 'vue'
  import { AdaptBox, MultArea } from 'yms-ui'
  import App from './App.vue'

  Vue.component(AdaptBox.name, AdaptBox)
  Vue.component(MultArea.name, MultArea)
  /* 或写为
  * Vue.use(Button)
  * Vue.use(Select)
  */

  new Vue({
    el: '#app',
    render: h => h(App)
  })
